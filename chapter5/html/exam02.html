public enum LoginResult {
    SUCCESS, 
    FAIL_ID, 
    FAIL_PASSWORD
}

// 사용 예제
LoginResult result = LoginResult.FAIL_PASSWORD;

    // 성공 처리
} else if (result == LoginResult.FAIL_ID) {
    // ID 실패 처리
}

🐍 Python Enum

from enum import Enum, auto

class LoginResult(Enum):
    SUCCESS = auto()
    FAIL_ID = auto()
    FAIL_PASSWORD = auto()
    
    def get_message(self):
        messages = {
            LoginResult.SUCCESS: "로그인 성공",
            LoginResult.FAIL_ID: "ID 오류",
            LoginResult.FAIL_PASSWORD: "PW 오류"
        }
        return messages.get(self)

⚡ TypeScript Enum

enum LoginResult {
    SUCCESS = "SUCCESS",
    FAIL_ID = "FAIL_ID",
    FAIL_PASSWORD = "FAIL_PASSWORD"
}

function getMessage(result: LoginResult): string {
    const messages: Record<LoginResult, string> = {
        [LoginResult.SUCCESS]: "로그인 성공",
        [LoginResult.FAIL_ID]: "ID 오류",
        [LoginResult.FAIL_PASSWORD]: "PW 오류"
    };
    return messages[result];
}
